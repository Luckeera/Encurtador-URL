def process_file(file_path, root_directory):
    """Processa um arquivo individual, lÃª o conteÃºdo e formata a saÃ­da."""
    result = []
    # ObtÃ©m o caminho relativo ao diretÃ³rio raiz do projeto
    relative_path = os.path.relpath(file_path, root_directory)

    result.append(f"\n{'â”€' * 80}")
    result.append(f"ğŸ·ï¸ ARQUIVO: {relative_path}")
    # ObtÃ©m o diretÃ³rio relativo
    relative_dir = os.path.dirname(relative_path)
    result.append(f"ğŸ“ LOCALIZAÃ‡ÃƒO: {relative_dir if relative_dir else 'raiz'}")
    result.append(f"{'â”€' * 80}")

    try:
        # Tenta abrir e ler o arquivo com codificaÃ§Ã£o UTF-8
        with open(file_path, 'r', encoding='utf-8') as f:
            content = f.read().strip() # Remove espaÃ§os em branco do inÃ­cio/fim

        # Verificar tamanho do conteÃºdo (pode usar len(content) ou len(content.splitlines()))
        # Usar len(content) para um limite de tamanho em bytes/caracteres
        # Usar len(content.splitlines()) para um limite de linhas
        lines = content.splitlines() # Divide em linhas para contar e limitar
        line_limit = 100 # Limite de linhas para arquivos grandes
        char_limit = 15000 # Limite de caracteres (aprox. 15KB)

        if len(content) > char_limit and len(lines) > line_limit:
             result.append(f"âš ï¸ ARQUIVO GRANDE ({len(lines)} linhas, ~{len(content)} caracteres) - MOSTRANDO PRIMEIRAS {line_limit} LINHAS")
             result.append('\n'.join(lines[:line_limit]))
             result.append(f"\n... [RESTANTE OMITIDO - {len(lines)-line_limit} linhas restantes]")
        elif len(content) == 0:
            result.append("ğŸ“ [ARQUIVO VAZIO]")
        else:
            result.append(content) # Inclui o conteÃºdo completo se for pequeno

    except UnicodeDecodeError:
        # Captura erro se o arquivo nÃ£o for UTF-8 (provavelmente binÃ¡rio)
        result.append("âš ï¸ [ARQUIVO BINÃRIO OU COM CODIFICAÃ‡ÃƒO DIFERENTE - NÃƒO PODE SER EXIBIDO COMO TEXTO]")
    except FileNotFoundError:
         # Embora os.walk encontre, Ã© uma boa prÃ¡tica ter este catch
         result.append("âŒ [ARQUIVO NÃƒO ENCONTRADO DURANTE O PROCESSAMENTO]")
    except Exception as e:
        # Captura quaisquer outros erros durante a leitura do arquivo
        result.append(f"âŒ [ERRO AO LER ARQUIVO: {e}]")

    return result
